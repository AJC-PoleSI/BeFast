   import { z } from "zod"

   export const profileSchema = z.object({
     prenom: z.string().min(1, "Le prenom est requis"),
     nom: z.string().min(1, "Le nom est requis"),
     portable: z.string().optional().or(z.literal("")),

     adresse: z.string().optional().or(z.literal("")),
     ville: z.string().optional().or(z.literal("")),
     code_postal: z.string().optional().or(z.literal("")),
     pole: z.string().optional().or(z.literal("")),
   })

   export type ProfileFormValues = z.infer<typeof profileSchema>

   export const sensitiveFieldSchema = z.object({
     field: z.enum(["nss", "iban"]),
     value: z.string().min(1, "Ce champ est requis"),
     confirmation: z.string().min(1, "La confirmation est requise"),
   }).refine(d => d.value === d.confirmation, {
     message: "Les valeurs ne correspondent pas.",
     path: ["confirmation"],
   })

   export const nssSchema = z.string().regex(
     /^[12]\d{2}(0[1-9]|1[0-2])\d{5}\d{2}$/,
     "Le numero de securite sociale n'est pas valide."
   )

   export const ibanSchema = z.string().regex(
     /^FR\d{2}\s?\d{4}\s?\d{4}\s?\d{4}\s?\d{4}\s?\d{4}\s?\d{3}$/,
     "L'IBAN n'est pas valide."
   )

   export const MAX_FILE_SIZE = 10 * 1024 * 1024 // 10 Mo
   export const ACCEPTED_FILE_TYPES = ["image/jpeg", "image/png", "image/webp", "application/pdf"]

     carte_etudiante: "Carte etudiante",
     carte_vitale: "Carte vitale",
     preuve_lydia: "Preuve Lydia",
     rib: "RIB",
   }

1. **`app/api/profil/route.ts`** — PATCH handler for basic profile info (PROF-01, PROF-05):
   - `import "server-only"` at top
   - Get user via `createClient()` from `@/lib/supabase/server`, return 401 if null
   - Parse body, validate with `profileSchema` from `@/app/(dashboard)/dashboard/profil/_lib/schemas`
   - Support optional `targetUserId` query param: if present AND current user is admin (check profils_types.slug === 'administrateur'), update that user instead (PROF-06)
   - If targetUserId present and user is NOT admin, return 403
   - Update personnes table via supabase client, return updated row
   - Return 200 with `{ success: true, data: updatedProfile }`

2. **`app/api/profil/sensitive/route.ts`** — POST handler for NSS/IBAN encryption (PROF-02, PROF-03, PROF-04, SEC-02):
   - `import "server-only"` at top
   - Auth check (401 if no user)
   - Parse body with `sensitiveFieldSchema`
   - Additional validation: if field === "nss", validate value with `nssSchema`; if field === "iban", validate value with `ibanSchema`
   - SECURITY: Only allow user to update their OWN sensitive fields (use auth.uid() only, never accept targetUserId — per 02-CONTEXT.md admin cannot edit others' sensitive data)
   - Call `encryptToString(value)` from `@/lib/encryption`
   - Use `createAdminClient()` to update the column (`nss_encrypted` or `iban_encrypted`) in personnes
   - Return `{ success: true }`

3. **`app/api/profil/avatar/route.ts`** — POST handler for avatar upload:
   - Auth check
   - Accept FormData with "file" field
   - Validate: image/* only, max 2 Mo (2 * 1024 * 1024)
   - Upload to `avatars` bucket path `{userId}/avatar.{ext}` with upsert:true via `createAdminClient()`
   - Get public URL via `admin.storage.from('avatars').getPublicUrl(filePath)`
   - Update `personnes.avatar_url` with the public URL
   - Return `{ success: true, url: publicUrl }`

4. **`app/api/profil/documents/route.ts`** — POST (upload) + GET (list) for documents (DOCS-01, DOCS-04, SEC-03):
   - **GET**: Auth check, query `documents_personnes` where personne_id = user.id (RLS handles this too), return array
   - Support optional `targetUserId` query param for admin view: if present, verify admin role, then query that user's docs
   - **POST**: Auth check, FormData with "file" and "docType" fields
   - Validate docType in VALID_DOC_TYPES, file size <= MAX_FILE_SIZE, mime_type in ACCEPTED_FILE_TYPES
   - Upload to `documents-personnes` bucket path `{userId}/{docType}/document.{ext}` with upsert:true via `createAdminClient()`
   - Upsert into `documents_personnes` table (personne_id, type, file_path, file_name, file_size, mime_type) using admin client
   - Return `{ success: true, document: insertedRow }`

5. **`app/api/profil/documents/[docId]/route.ts`** — DELETE handler (DOCS-03):
   - Auth check
   - Fetch document by id from `documents_personnes`, verify personne_id === user.id (ownership)
   - Delete file from Storage via `admin.storage.from('documents-personnes').remove([doc.file_path])`
   - Delete row from `documents_personnes` via admin client
   - Return `{ success: true }`

6. **`app/api/profil/documents/signed-url/route.ts`** — POST handler for signed URL (DOCS-05):
   - Auth check
   - Accept `{ filePath }` in body
   - Verify the file belongs to user (query documents_personnes by file_path and personne_id) OR user is admin
   - Generate signed URL: `admin.storage.from('documents-personnes').createSignedUrl(filePath, 3600)` — 3600 seconds = 1 hour
   - Return `{ url: signedUrl }`

ALL routes: import `"server-only"` at top, first lines always auth check with 401 return, use `NextResponse.json()` for responses, wrap in try/catch with 500 fallback `"Une erreur est survenue."`.
